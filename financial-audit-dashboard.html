SQL Query Used — Anomaly Detection Logic
-- Benford's Law first-digit distribution check
SELECT
  SUBSTR(CAST(ABS(amount) AS TEXT), 1, 1) AS first_digit,
  COUNT(*) AS actual_count,
  COUNT(*) * 100.0 / SUM(COUNT(*)) OVER() AS actual_pct,
  LOG10(1 + 1.0/CAST(SUBSTR(CAST(ABS(amount) AS TEXT),1,1) AS FLOAT))*100 AS benford_pct,
  ABS(COUNT(*)*100.0/SUM(COUNT(*))OVER() - LOG10(1+1.0/CAST(SUBSTR(CAST(ABS(amount) AS TEXT),1,1) AS FLOAT))*100) AS deviation
FROM transactions
WHERE transaction_date BETWEEN '2024-01-01' AND '2024-12-31'
  AND amount > 0
GROUP BY first_digit
HAVING deviation > 5.0
ORDER BY first_digit;
